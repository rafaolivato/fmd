# Rodar o servidor

npx ts-node src/server.ts

# Rodar o front-endereco

npm run dev

# Na pasta de Backend 

npx prisma db seed

# 1. Valide o schema
npx prisma validate

# 2. Gere o cliente
npx prisma generate

# 3. Sincronize com o banco
npx prisma db push
# OU se usar migrações:
npx prisma migrate dev --name sync_schema


# 4. Quando aparecer erro interno
npx prisma migrate dev

####Lembrete importante usar o prisma studio é bem fácil para mexer no banco

npx prisma studio

# 5. Para rodar o seed inicial 

npm run seed //user (admin)
npm run seed:medicamentos


# 6. Instalar dependências (apenas se for a primeira vez)
npm install 

# 7. Para deletar o banco de dados ( sempre lembra de rodar o npm minstall antes)
npx prisma migrate reset
